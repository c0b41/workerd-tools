addEventListener("fetch", (event) => {
    const request = new Request(event.request);
    const url = new URL(request.url);
    const LoopBackPath = BINDING_LOOPBACK_PATH ? BINDING_LOOPBACK_PATH : '';
    url.pathname = `${LoopBackPath}/${BINDING_PLUGIN}/${encodeURIComponent(request.url)}`;
    if (globalThis.BINDING_NAMESPACE !== undefined) request.headers.set("SRK-ID", BINDING_NAMESPACE);

    event.respondWith(BINDING_LOOPBACK.fetch(url, request));
});
